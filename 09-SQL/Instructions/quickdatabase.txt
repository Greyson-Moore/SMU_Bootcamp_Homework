dept_no varchar(15) pk
dept_name varchar(30)

dept_emp
-
emp_no int FK >- Employees.emp_no
dept_no varchar(15) FK >- Departments.dept_no

dept_manager
-
dept_no varchar(15) FK >- Departments.dept_no
emp_no int FK >- Employees.emp_no

emp_no int pk
emp_title varchar(15) FK >- Titles.title_id

sex varchar(10)
hire_date varchar(30)

Salaries
-
